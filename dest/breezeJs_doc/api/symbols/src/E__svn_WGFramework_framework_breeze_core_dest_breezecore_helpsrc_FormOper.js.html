145     _api.checkData = function(checkData,checkMetadata){
146     	//定义一个全值
147     	var __data=checkData;
148     	//这时一个递归函数
149     	function checkValueFun(checkObj,checkDesc){    		
150 			for(var name in checkDesc){
151 				if (!checkDesc.hasOwnProperty(name)){
152 					continue;
153 				}
154 				var checkValue = checkObj[name] || "";
155 				var descObj = checkDesc[name];
156 
157 				//过滤不需要校验的对象
158 				if (descObj.type == "Select" ||descObj.type == "MultSelect"||descObj.type == "hidden"||descObj.type == "CheckBox"||descObj.type == "Radio"||descObj.type == "Json"){
159 					continue;
160 				}
161 				//检查是否需要校验
162 				if (!descObj.valueRange || (descObj.type == "List" && descObj.valueRange.length==0)){
163 					continue;
164 				}
165 				for (var i=0;i<descObj.valueRange.length;i++){
166 					if (descObj.type == "List" && i==0){
167 						for (var j=0;j<checkValue.length;j++){
168 							if (checkValueFun(checkValue[j],descObj.valueRange[i]) == null){
169 								return null;
170 							}
171 						}
172 						continue;
173 					}
174 					var checkOne = descObj.valueRange[i];
175 
176 					//2013-11-23罗光瑜修改，这里要重新使用会原来的功能，当然function是一个字符串，同样json也是一个字符串
177 					//因为标准的json里面如果作为通信传输的话，这两个对象（function和json）是不能被传输的，所以都转成字符串了
178 					if(checkOne.checkers ){
179 						if (/function/i.test(typeof(checkOne.checkers)) || /^function/i.test(checkOne.checkers)){									
180 							var funCheck = checkOne.checkers;
181 							if (/string/i.test(typeof(funCheck))){
182 								funCheck = eval ("("+checkOne.checkers+")");
183 							}
184 							//这是一个函数
185 							if (!funCheck(checkValue,__data)){
186 								//没有通过
187 								alert(checkOne.failTips);
188 								return null;
189 							}
190 						}else{
191 							var cReg = checkOne.checkers;
192 							if (/string/i.test(typeof(cReg))){
193 								cReg = new RegExp(checkOne.checkers,"i");
194 							}
195 							//2013-12-23日罗光瑜添加，要保证这是一个正则表达式									
196 							if (cReg.test && !cReg.test(checkValue)){
197 								alert(checkOne.failTips);
198 								return null;
199 							}
200 						}
201 					}
202 				}
203 			}
204 			return checkObj;
205 		}
206 		return checkValueFun(checkData,checkMetadata);
