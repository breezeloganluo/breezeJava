 20         function trim_output() {
 21             while (output.length && (output[output.length - 1] === ' ' || output[output.length - 1] === indent_string)) {
 22                 output.pop();
 23             }

 26         function print_newline(ignore_repeated) {
 27             ignore_repeated = typeof ignore_repeated === 'undefined' ? true : ignore_repeated;
 28 
 29             trim_output();
 30 
 31             if (!output.length) {
 32                 return; // no newline on start of file
 33             }
 34 
 35             if (output[output.length - 1] !== "\n" || !ignore_repeated) {
 36                 output.push("\n");
 37             }
 38             for (var i = 0; i < indent_level; i++) {
 39                 output.push(indent_string);
 40             }

 45         function print_space() {
 46             var last_output = output.length ? output[output.length - 1] : ' ';
 47             if (last_output !== ' ' && last_output !== '\n' && last_output !== indent_string) { // prevent occassional duplicate space
 48                 output.push(' ');
 49             }

 88         function in_array(what, arr) {
 89             for (var i = 0; i < arr.length; i++) {
 90                 if (arr[i] === what) {
 91                     return true;
 92                 }
 93             }

 99         function get_next_token() {
100             var n_newlines = 0;
101             var c = '';
102 
103             do {
104                 if (parser_pos >= input.length) {
105                     return ['', 'TK_EOF'];
106                 }
107                 c = input.charAt(parser_pos);
108 
109                 parser_pos += 1;
110                 if (c === "\n") {
111                     n_newlines += 1;
112                 }
113             }
114             while (in_array(c, whitespace));
115 
116             if (n_newlines > 1) {
117                 for (var i = 0; i < 2; i++) {
118                     print_newline(i === 0);
119                 }
120             }
121             var wanted_newline = (n_newlines === 1);
122 
123 
124             if (in_array(c, wordchar)) {
125                 if (parser_pos < input.length) {
126                     while (in_array(input.charAt(parser_pos), wordchar)) {
127                         c += input.charAt(parser_pos);
128                         parser_pos += 1;
129                         if (parser_pos === input.length) {
130                             break;
131                         }
132                     }
133                 }
134 
135                 // small and surprisingly unugly hack for 1E-10 representation
136                 if (parser_pos !== input.length && c.match(/^[0-9]+[Ee]$/) && input.charAt(parser_pos) === '-') {
137                     parser_pos += 1;
138 
139                     var t = get_next_token(parser_pos);
140                     c += '-' + t[0];
141                     return [c, 'TK_WORD'];
142                 }
143 
144                 if (c === 'in') { // hack for 'in' operator
145                     return [c, 'TK_OPERATOR'];
146                 }
147                 return [c, 'TK_WORD'];
148             }
149 
150             if (c === '(' || c === '[') {
151                 return [c, 'TK_START_EXPR'];
152             }
153 
154             if (c === ')' || c === ']') {
155                 return [c, 'TK_END_EXPR'];
156             }
157 
158             if (c === '{') {
159                 return [c, 'TK_START_BLOCK'];
160             }
161 
162             if (c === '}') {
163                 return [c, 'TK_END_BLOCK'];
164             }
165 
166             if (c === ';') {
167                 return [c, 'TK_END_COMMAND'];
168             }
169 
170             if (c === '/') {
171                 var comment = '';
172                 // peek for comment /* ... */
173                 if (input.charAt(parser_pos) === '*') {
174                     parser_pos += 1;
175                     if (parser_pos < input.length) {
176                         while (!(input.charAt(parser_pos) === '*' && input.charAt(parser_pos + 1) && input.charAt(parser_pos + 1) === '/') && parser_pos < input.length) {
177                             comment += input.charAt(parser_pos);
178                             parser_pos += 1;
179                             if (parser_pos >= input.length) {
180                                 break;
181                             }
182                         }
183                     }
184                     parser_pos += 2;
185                     return ['/*' + comment + '*/', 'TK_BLOCK_COMMENT'];
186                 }
187                 // peek for comment // ...
188                 if (input.charAt(parser_pos) === '/') {
189                     comment = c;
190                     while (input.charAt(parser_pos) !== "\x0d" && input.charAt(parser_pos) !== "\x0a") {
191                         comment += input.charAt(parser_pos);
192                         parser_pos += 1;
193                         if (parser_pos >= input.length) {
194                             break;
195                         }
196                     }
197                     parser_pos += 1;
198                     if (wanted_newline) {
199                         print_newline();
200                     }
201                     return [comment, 'TK_COMMENT'];
202                 }
203 
204             }
205 
206             if (c === "'" || // string
207                 c === '"' || // string
208                 (c === '/' &&
209                     ((last_type === 'TK_WORD' && last_text === 'return') || (last_type === 'TK_START_EXPR' || last_type === 'TK_END_BLOCK' || last_type === 'TK_OPERATOR' || last_type === 'TK_EOF' || last_type === 'TK_END_COMMAND')))) { // regexp
210                 var sep = c;
211                 var esc = false;
212                 c = '';
213 
214                 if (parser_pos < input.length) {
215 
216                     while (esc || input.charAt(parser_pos) !== sep) {
217                         c += input.charAt(parser_pos);
218                         if (!esc) {
219                             esc = input.charAt(parser_pos) === '\\';
220                         } else {
221                             esc = false;
222                         }
223                         parser_pos += 1;
224                         if (parser_pos >= input.length) {
225                             break;
226                         }
227                     }
228 
229                 }
230 
231                 parser_pos += 1;
232                 if (last_type === 'TK_END_COMMAND') {
233                     print_newline();
234                 }
235                 return [sep + c + sep, 'TK_STRING'];
236             }
237 
238             if (in_array(c, punct)) {
239                 while (parser_pos < input.length && in_array(c + input.charAt(parser_pos), punct)) {
240                     c += input.charAt(parser_pos);
241                     parser_pos += 1;
242                     if (parser_pos >= input.length) {
243                         break;
244                     }
245                 }
246                 return [c, 'TK_OPERATOR'];
247             }
248 
249             return [c, 'TK_UNKNOWN'];

383                     } else if (last_type === 'TK_END_COMMAND' && current_mode === 'EXPRESSION') {
384                         prefix = 'SPACE';
385                     } else if (last_type === 'TK_WORD') {

387                     } else if (last_type === 'TK_START_BLOCK') {

389                     } else if (last_type === 'TK_END_EXPR') {
390                         print_space();

392                     }
393

396                     } else if (in_array(token_text, line_starters) || prefix === 'NEWLINE') {
397                         if (last_text === 'else') {
398                             // no need to force newline on else break
399                             print_space();
400                         } else if ((last_type === 'TK_START_EXPR' || last_text === '=') && token_text === 'function') {
401                             // no need to force newline on 'function': (function
402                             // DONOTHING
403                         } else if (last_type === 'TK_WORD' && (last_text === 'return' || last_text === 'throw')) {
404                             // no newline between 'return nnn'
405                             print_space();
406                         } else if (last_type !== 'TK_END_EXPR') {
407                             if ((last_type !== 'TK_START_EXPR' || token_text !== 'var') && last_text !== ':') {
408                                 // no need to force newline on 'var': for (var x = 0...)
409                                 if (token_text === 'if' && last_type === 'TK_WORD' && last_word === 'else') {
410                                     // no newline for } else if {
411                                     print_space();
412                                 } else {
413                                     print_newline();
414                                 }
415                             }
416                         } else {
417                             if (in_array(token_text, line_starters) && last_text !== ')') {
418                                 print_newline();
419                             }
420                         }
421                     } else if (prefix === 'SPACE') {
422                         print_space();
423                     }
424                     print_token();
425                     last_word = token_text;
426 
427                     if (token_text === 'var') {
428                         var_line = true;
429                         var_line_tainted = false;
430                     }
431 
432                     break;
433 
434                 case 'TK_END_COMMAND':
435

437                     var_line = false;
438                     break;
439 
440                 case 'TK_STRING':
441 
442                     if (last_type === 'TK_START_BLOCK' || last_type === 'TK_END_BLOCK') {
443                         print_newline();
444                     } else if (last_type === 'TK_WORD') {
445                         print_space();
446                     }
447                     print_token();
448                     break;
449 
450                 case 'TK_OPERATOR':
451 
452                     var start_delim = true;
453                     var end_delim = true;
454                     if (var_line && token_text !== ',') {
455                         var_line_tainted = true;
456                         if (token_text === ':') {
457                             var_line = false;
458                         }
459                     }
460 
461                     if (token_text === ':' && in_case) {
462                         print_token(); // colon really asks for separate treatment
463                         print_newline();
464                         break;
465                     }
466 
467                     in_case = false;
468 
469                     if (token_text === ',') {
470                         if (var_line) {
471                             if (var_line_tainted) {
472                                 print_token();
473                                 print_newline();
474                                 var_line_tainted = false;
475                             } else {
476                                 print_token();
477                                 print_space();
478                             }
479                         } else if (last_type === 'TK_END_BLOCK') {
480                             print_token();
481                             print_newline();
482                         } else {
483                             if (current_mode === 'BLOCK') {
484                                 print_token();
485                                 print_newline();
486                             } else {
487                                 // EXPR od DO_BLOCK
488                                 print_token();
489                                 print_space();
490                             }
491                         }
492                         break;
493                     } else if (token_text === '--' || token_text === '++') { // unary operators special case
494                         if (last_text === ';') {
495                             // space for (;; ++i)
496                             start_delim = true;
497                             end_delim = false;
498                         } else {
499                             start_delim = false;
500                             end_delim = false;
501                         }
502                     } else if (token_text === '!' && last_type === 'TK_START_EXPR') {
503                         // special case handling: if (!a)
504                         start_delim = false;
505                         end_delim = false;
506                     } else if (last_type === 'TK_OPERATOR') {
507                         start_delim = false;
508                         end_delim = false;
509                     } else if (last_type === 'TK_END_EXPR') {
510                         start_delim = true;
511                         end_delim = true;
512                     } else if (token_text === '.') {
513                         // decimal digits or object.property
514                         start_delim = false;
515                         end_delim = false;
516 
517                     } else if (token_text === ':') {

519                         // can't differentiate ternary op, so for now it's a ? b: c; without space before colon

521                             // a little help for ternary a ? 1 : 0;
522                             start_delim = true;
523                         } else {
524                             start_delim = false;
525                         }
526                     }
527                     if (start_delim) {
528                         print_space();
529                     }
530 
531                     print_token();
532 
533                     if (end_delim) {
534                         print_space();
535                     }
536                     break;
537 
538                 case 'TK_BLOCK_COMMENT':

542 					var my_space = "";
543 					for (var i = 0; i < indent_level; i++) {
544 						my_space+=indent_string;
545 					}
546 					if (my_space != ""){

548 						for(var i=1;i<myCmmArr.length;i++){

555                     break;
556 
557                 case 'TK_COMMENT':
558

560                     print_space();
561                     print_token();
562                     print_newline();
563                     break;
564 
565                 case 'TK_UNKNOWN':
566                     print_token();
567                     break;
568             }
569 
570             last_type = token_type;
571             last_text = token_text;
572         }
573         return output.join('');
