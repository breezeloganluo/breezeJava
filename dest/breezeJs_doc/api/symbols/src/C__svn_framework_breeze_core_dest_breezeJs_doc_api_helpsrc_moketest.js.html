1 /** 
  2 * @fileOverview FW配套使用的核心默认函数lang 
  3 * @author <a href="http://www.wgfly.com">Logan</a> 
  4 * @version 0.1
  5 * @version 0.01 罗光瑜修改 getParameter函数
  6 */ 
  7 
  8 /**
  9 * @namespace
 10 * @author Logan 
 11 * @name moketest
 12 * @description  FW的核心基本扩展 
 13 */ 
 14 define(function(require, exports, module) {
 15 	//用函数自身做域名承载对象
 16 	var _result = function(fw){
 17 		fw.use(_result);
 18 	}
 19 	
 20 	
 21 	
 22 	_result.getDomain = function(){
 23 		return "moketest";
 24 	}
 25 	
 26 	
 27 	
 28 	FW && FW.use(_result);
 29 	/**
 30 	* @function
 31 	* @memberOf moketest
 32 	* @name getMokeData
 33 	* @param {Object} app
 34 	* @desctiption 远程去获取模拟数据，参数从threadSignal=xx来获取对应的数据对象
 35 	*/
 36 	_result.getMokeData = function(app){
 37 		//获取名字
 38 		var name = app.gadgetName;
 39 		
 40 		//获取方法名
 41 		var funName = "";
 42 		var method = FW.getCaller(1);
 43 		for (var n in app){
 44 			if (app[n] == method){
 45 				funName = n;
 46 			}
 47 		}
 48 		if (funName == ""){
 49 			for (var n in app.private){
 50 				if (app.private[n] == method){
 51 					funName = n;
 52 				}
 53 			}
 54 		}
 55 		
 56 		if (funName == ""){
 57 			for (var n in app.FireEvent){
 58 				if (app.FireEvent[n] == method){
 59 					funName = n;
 60 				}
 61 			}
 62 		}
 63 		
 64 		//获取方法参数
 65 		var args = [];
 66 		if (method.arguments && method.arguments.length > 0){
 67 			for(var i=0;i<method.arguments.length;i++){
 68 				if(/function/i.test(method.arguments[i])){
 69 					continue;
 70 				}
 71 				args.push(method.arguments[i]);
 72 			}
 73 		}
 74 		
 75 	    //获取url文件
 76 		var myurl = window.location.toString();
 77 		var url = myurl.replace(/^http[s]?:\/\/[^\/]+/i, "");
 78 		//去掉尾部?部分
 79 		url = url.replace(/[#\?][\s\S]*$/i, "");
 80 		//去掉应用名部分
 81 		if (/undefined/.test(typeof(Cfg))){
 82 			alert("Cfg.baseUrl必须被定义，在/config/config.jsp中定义按照js方式加载");
 83 			return;
 84 		}
 85 		if (!/undefined/.test(typeof(Cfg)) && url.indexOf(Cfg.baseUrl) == 0) {
 86 			url = url.substr(Cfg.baseUrl.length);
 87 		}else{
 88 			url = url.replace(/^.*(\/page\/)/i,function(a,b,c,d,e){
 89 				return b;
 90 			})
 91 		}
 92 		//变成可用符号
 93 		url=("/"+url).replace(/[\\\/]+/ig,"_");
 94 		
 95 		var result = FW.doServer("getMokeData","moke",{
 96 			url:url,
 97 			gadgetName:name,
 98 			funName:funName,
 99 			input:args
100 		},null,null,Cfg.baseUrl+"/breeze/framework/jsp/BreezeFW.jsp");
101 		!/undefined/i.test(typeof(console)) && console.log(result);
102 		if (result.code != 0 || result.data == null){
103 			return{};
104 		}
105 		return result.data;
106 	}
107 
108 	return _result;
109 });
