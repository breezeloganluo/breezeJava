 32 	 * @function
 33 	 * @memberOf lang
 34 	 * @name getParameter
 35 	 * @description 从url中获取参数
 36 	 * @param {String}pName 参数名
 37 	 * @returns 参数名对应的参数值
 38 	 */
 39 	_result.getParameter=function(__pName){
 40 		var url = window.location.toString();
 41 		//2014-05-28 罗光瑜修改 某些IE浏览器url地址是不进行中文转换的，这时取参数正则要修改
 42 		//2014年8月13日 16:12:47 程明剑 添加支持type参数
 43 		var execResult = (new RegExp(__pName+"=([^&#]+)|"+__pName+"_([^*&.-]+)")).exec(url);
 44 		if (execResult){
 45 			//2014年1月14日 20:46:56 程明剑 添加URL解密
 46 			//2014年8月13日 16:13:04 程明剑 判断是正常参数还是type参数
 47 			var result = execResult[1] == null ?execResult[2]:execResult[1];
 48 			return decodeURIComponent(result);
 49 		}
 50 		return null;
 51 	};

 53 	/**
 54 	 * @function
 55 	 * @memberOf lang
 56 	 * @name save
 57 	 * @desctiption 本地存储,存
 58 	 * @param {String} name 存储名称
 59 	 * @param {Object} data 被存储的对象，如果为null，则是把name删除掉
 60 	 * @param {boolean} [isPermanent] 是否永久性存储，true是永久存储，false或默认不填是会话存储
 61 	 * @see load
 62 	 * @example
 63 	 * FW.use().save("abc",{a:'hello'});//存储
 64 	 * FW.use().save("abc",null);//删除
 65 	 * 
 66 	 */
 67 	_result.save=function(name,data,isPermanent){
 68 		if(!window.localStorage || !window.sessionStorage ){
 69 			seajs.log("browser not supprt window.loalStorage!");
 70 			return;
 71 		}
 72 		var storage = (isPermanent)?localStorage:sessionStorage;
 73 		if (data){
 74 			storage.setItem(name,APICtr.JsonAPI.toJSONString(data));					
 75 		}else{
 76 			//如果是null，表示删除
 77 			storage.removeItem(name);
 78 		}
 79 	};

191 			for(var i in object){
192 				if(object[i]!=null&&typeof(object[i])=="object"){
193 					deleteNull(object[i]);
194 				}
195 				if(object[i]==null){

198 				}
199 			}
200 		}

202 		return APICtr.JsonAPI.toJSONString(object);

626 	/**
627 	*深度拷贝，这个方法用于深度的拷贝一个对象
628 	* @memberOf lang
629 	*@name deepCopy
630 	*@param __srcObj 原始的对象
631 	*@param __destObj 拷贝到的目标对象
632 	*@param __filterFun 只是一个拷贝过程中回调的过滤函数，回调的输入参数有如下：
633 	*                  srcMemberName当前要拷贝的成员名,path 路径名,srcObj上一级父亲名，destObj上一级父亲名,
634 	* 					即，使用srcObj[srcMemberName]就可以获取到当前正在要拷贝的对象了
635 	*                  返回值true这个成员继续拷贝，否则不拷贝了
636 	*/
637 	_result.deepCopy = function(__srcObj, __destObj, __filterFun,_allPath) {
638 		if (_allPath == null){
639 			_allPath = "";
640 		}
641 		for (var name in __srcObj) {
642 			if (!__srcObj.hasOwnProperty(name)) {
643 				continue;
644 			}
645 			
646 			//先用传入的filter进行过滤
647 			var allPath = "";
648 			if (_allPath != ""){
649 				allPath = _allPath + "." + name;
650 			}else{
651 				allPath = _allPath + name;
652 			}
653 			if (__filterFun && !__filterFun(name,allPath,__srcObj,__destObj)){
654 				continue;
655 			}
656 			var oo = __srcObj[name];
657 			//2013-12-06 罗光瑜修改 如果是null那么直接跳过这个成员
658 			if (oo == null){
659 				continue;
660 			}
661 			if (/function/i.test(typeof(oo))) {
662 				//2013-11-15日lgy修改，函数也要进行覆盖性判断，如果目标子类存在就不用父类的方法
663 				if (__destObj[name] == null){
664 					__destObj[name] = oo;
665 				}
666 				continue;
667 			}
668 			if (/Object/i.test(typeof(oo))) {
669 				if (__destObj[name] == null) {
670 					//lgy 2013-12-06 如果目标不存在，因为这里是对象类型，所以不能直接赋值，否则不是深度拷贝
671 					//判断数组类型
672 					if (oo instanceof Array){
673 						__destObj[name] = [];
674 					}else{
675 						__destObj[name] = {};
676 					}
677 				}
678 				//递归调用对对象
679 				
680 				_result.deepCopy(oo, __destObj[name], __filterFun,allPath);
681 				continue;
682 			}
683 			//lgy 2013-11-13 子类如果存在对应的值就不能用父类覆盖。这里oo是父类的值
684 			if (__destObj[name] == null){
685 				__destObj[name] = oo;
686 			}					
687 		}
688 	}

690 	/**
691 	*根据字符串对象路径设置对象值，如果设置的过程中对象成员不存在就会被创建<br/>
692 	*例如传入的是var data = {};<br/>
693 	* setObjStrData(data,"a.b[0][1].c[][].d","ok");就会把ok设置到data.a.b[0][1].c[0][0].d下面
694 	* @memberOf lang
695 	*@name setObjStrData
696 	*@param data 要设置的对象的根
697 	*@param str 拷贝到的目标对象
698 	*@param value 实际要设置的值
699 	*/
700 	_result.setObjStrData = function(data,str,value){
701 	    //先声明一个设置值的函数，用于判断被设置对象原来的值是否存在，如果存在则变成数组
702 		var setValue = function(obj,key,value){
703 			if (obj[key] == null){
704 				obj[key] = value;
705 				return;
706 			}
707 			if (obj[key] instanceof Array){
708 				obj[key].push(value);
709 				return;
710 			}
711 			
712 			var tmp = obj[key];
713 			obj[key] = [];
714 			obj[key].push(tmp);
715 			obj[key].push(value);
716 		}
717 		
718 		
719 		
720 		
721 	    //用正则分解第一项
722 		var execResult = /(\w+)([\.\[\d\]]*)([\s\S]*)/.exec(str);
723 		if (execResult == null){
724 			return;
725 		}
726 		var key = execResult[1];
727 		var after = execResult[2];
728 		var next = execResult[3];
729 		
730 		//如果没有后缀，那么next肯定也不存在，赋值后直接返回
731 		if (after == null || after==""){
732 			setValue(data,key,value);
733 			return;
734 		}
735 		
736 		//如果仅仅是.表示这个是对象，还有后续部分的
737 		if (after == "."){
738 			//应该有后续的没有后续一定是出错了
739 		    if (next == null){
740 				return;
741 			}
742 		    if (data[key] == null){
743 			   data[key] = {};
744 			}
745 			_result.setObjStrData(data[key],next,value);
746 			return;
747 		}
748 		
749 		//下面处理最复杂的[]情况
750 		var curStr = after;
751 		if (!data[key]){
752 			data[key] = [];
753 		}
754 		var curData = data[key];
755 		while(true){
756 		    execResult = /\[(\d*)\](\.?)([\s\S]*)/i.exec(curStr);
757 			//没有匹配到说明出错了
758 			if (execResult == null){
759 			   return;
760 			}
761 
762 			var idx = execResult[1];
763 			var isObj = (execResult[2] != null && execResult[2]!="");
764 			var isArr = (execResult[3] != null && execResult[3]!="");
765             //数组无下标，补充下标
766 			if (idx == null || idx == ""){
767 				idx = curData.length;
768 			}
769 			//本身无值的情况
770 			if (!curData[idx]){
771 				//是对象，递归去
772 				if (isObj){
773 					curData[idx]={};
774 					_result.setObjStrData(curData[idx],next,value);
775 					return;
776 				}
777 				//如果是数组，继续循环
778 				if (isArr){
779 					curData[idx] = [];
780 					curData = curData[idx];
781 					curStr = execResult[3];
782 					continue;
783 				}
784 				//不是数组，也不是对象，那么当前就是要赋值了a[2]
785 				setValue(curData,idx,value);
786 				return;
787 			}
788 			//本身有值的情况
789 			else{
790 				//是对象情况,递归去
791 				if (isObj){
792 					_result.setObjStrData(curData[idx],next,value);
793 					return;
794 				}
795 				//如果是数组，继续循环
796 				if (isArr){
797 					curData = curData[idx];
798 					curStr = execResult[3];
799 					continue;
800 				}
801 				//直接赋值
802 				setValue(curData,idx,value);
803 				return;
804 			}
805 		}
806     };
