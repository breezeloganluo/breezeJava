  4     handleError: function( s, xhr, status, e ) 		{
  5     	// If a local callback was specified, fire it
  6 		if ( s.error ) {
  7 			s.error.call( s.context || s, xhr, status, e );
  8 		}
  9 
 10 		// Fire the global callback
 11 		if ( s.global ) {
 12 			(s.context ? jQuery(s.context) : jQuery.event).trigger( "ajaxError", [xhr, s, e] );
 13 		}
 14 	},

 66     ajaxFileUpload: function(s) {
 67         // TODO introduce global settings, allowing the client to modify them for all requests, not only timeout		
 68         s = jQuery.extend({}, jQuery.ajaxSettings, s);
 69         var id = new Date().getTime()        
 70 		var form = jQuery.createUploadForm(id, s.fileElementId, (typeof(s.data)=='undefined'?false:s.data));
 71 		var io = jQuery.createUploadIframe(id, s.secureuri);
 72 		var frameId = 'jUploadFrame' + id;
 73 		var formId = 'jUploadForm' + id;		
 74         // Watch for a new set of requests
 75         if ( s.global && ! jQuery.active++ )
 76 		{
 77 			jQuery.event.trigger( "ajaxStart" );
 78 		}            
 79         var requestDone = false;
 80         // Create the request object
 81         var xml = {}   
 82         if ( s.global )
 83             jQuery.event.trigger("ajaxSend", [xml, s]);
 84         // Wait for a response to come back
 85         var uploadCallback = function(isTimeout)
 86 		{			
 87 			var io = document.getElementById(frameId);
 88             try 
 89 			{				
 90 				if(io.contentWindow)
 91 				{
 92 					 xml.responseText = io.contentWindow.document.body?io.contentWindow.document.body.innerHTML:null;
 93                 	 xml.responseXML = io.contentWindow.document.XMLDocument?io.contentWindow.document.XMLDocument:io.contentWindow.document;
 94 					 
 95 				}else if(io.contentDocument)
 96 				{
 97 					 xml.responseText = io.contentDocument.document.body?io.contentDocument.document.body.innerHTML:null;
 98                 	xml.responseXML = io.contentDocument.document.XMLDocument?io.contentDocument.document.XMLDocument:io.contentDocument.document;
 99 				}						
100             }catch(e)
101 			{
102 				jQuery.handleError(s, xml, null, e);
103 			}
104             if ( xml || isTimeout == "timeout") 
105 			{				
106                 requestDone = true;
107                 var status;
108                 try {
109                     status = isTimeout != "timeout" ? "success" : "error";
110                     // Make sure that the request was successful or notmodified
111                     if ( status != "error" )
112 					{
113                         // process the data (runs the xml through httpData regardless of callback)
114                         var data = jQuery.uploadHttpData( xml, s.dataType );    
115                         // If a local callback was specified, fire it and pass it the data
116                         if ( s.success )
117                             s.success( data, status );
118     
119                         // Fire the global callback
120                         if( s.global )
121                             jQuery.event.trigger( "ajaxSuccess", [xml, s] );
122                     } else
123                         jQuery.handleError(s, xml, status);
124                 } catch(e) 
125 				{
126                     status = "error";
127                     jQuery.handleError(s, xml, status, e);
128                 }
129 
130                 // The request was completed
131                 if( s.global )
132                     jQuery.event.trigger( "ajaxComplete", [xml, s] );
133 
134                 // Handle the global AJAX counter
135                 if ( s.global && ! --jQuery.active )
136                     jQuery.event.trigger( "ajaxStop" );
137 
138                 // Process result
139                 if ( s.complete )
140                     s.complete(xml, status);
141 
142                 jQuery(io).unbind()
143 
144                 setTimeout(function()
145 									{	try 
146 										{
147 											jQuery(io).remove();
148 											jQuery(form).remove();	
149 											
150 										} catch(e) 
151 										{
152 											jQuery.handleError(s, xml, null, e);
153 										}									
154 
155 									}, 100)
156 
157                 xml = null
158 
159             }
160         }
161         // Timeout checker
162         if ( s.timeout > 0 ) 
163 		{
164             setTimeout(function(){
165                 // Check to see if the request is still happening
166                 if( !requestDone ) uploadCallback( "timeout" );
167             }, s.timeout);
168         }
169         try 
170 		{
171 
172 			var form = jQuery('#' + formId);
173 			jQuery(form).attr('action', s.url);
174 			jQuery(form).attr('method', 'POST');
175 			jQuery(form).attr('target', frameId);
176             if(form.encoding)
177 			{
178 				jQuery(form).attr('encoding', 'multipart/form-data');      			
179             }
180             else
181 			{	
182 				jQuery(form).attr('enctype', 'multipart/form-data');			
183             }			
184             jQuery(form).submit();
185 
186         } catch(e) 
187 		{			
188             jQuery.handleError(s, xml, null, e);
189         }
190 		
191 		jQuery('#' + frameId).load(uploadCallback	);
192         return {abort: function () {}};	
193 
194     },

196     uploadHttpData: function( r, type ) {
197         var data = !type;
198         data = type == "xml" || data ? r.responseXML : r.responseText;
199         // If the type is "script", eval it in global context
200    
201         if ( type == "script" )
202             jQuery.globalEval( data );
203         // Get the JavaScript object, if JSON is used.
204         if ( type == "json" )
205 
206         	//eval("data = \" "+data+" \" ");
207         	eval( "data = " + data );
208         // evaluate scripts within html
209         if ( type == "html" )
210             jQuery("<div>").html(data).evalScripts();
211 
212         return data;
213     }
