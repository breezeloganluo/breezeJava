507 		 * @param {Object} eNode 终止形状

513 		graph.createEdge = function(text,sNode,eNode,lineArr,linetype,endType,color){
514 			//整理一个画线折点
515 			var allPoint = [];
516 			var nextPoint = (lineArr && lineArr[0]) || eNode.point.center;
517 			var dx = nextPoint.x - sNode.point.center.x;
518 			var dy = nextPoint.y - sNode.point.center.y;
519 			//设定一个函数,输入是要比较的那个对象,和dx,dy,然后判断起始的那个对象,到底取哪个店,再push到allporint中
520 			var setPoint = function(settingNode,dx,dy){
521 				//判断第一个点是上下左右的哪一个
522 				if (dx >=0 && dy >=0){
523 					//右下角的
524 					if (dx > dy){
525 						//用右边
526 						allPoint.push(settingNode.point.right);
527 					}else{
528 						//用下面的
529 						allPoint.push(settingNode.point.down);
530 					}
531 				}else if (dx >= 0 && dy <= 0){
532 					//右上角的
533 					if (dx> -dy){
534 						allPoint.push(settingNode.point.right);
535 					}else{
536 						allPoint.push(settingNode.point.up);
537 					}
538 				}else if (dx <= 0 && dy <= 0){
539 					//左上角的
540 					if (-dx > -dy){
541 						allPoint.push(settingNode.point.left);
542 					}else{
543 						allPoint.push(settingNode.point.up);
544 					}
545 				}else{
546 					//就在左下角了
547 					if (-dx >= dy){
548 						allPoint.push(settingNode.point.left);
549 					}else{
550 						allPoint.push(settingNode.point.down);
551 					}
552 				}
553 			}
554 			//添加内容，如果dx,dy==0就默认是自己接自己，这种情况就作为自返处理
555 			if (dx == dy && dx == 0  && (lineArr == null || lineArr.length == null || lineArr.length == 0)){
556 				//自返情况
557 				var tmpPoint = {
558 					x : sNode.point.right.x,
559 					y : sNode.point.right.y-10
560 				}
561 				allPoint.push(tmpPoint);
562 				
563 				tmpPoint = {
564 					x:sNode.point.right.x + 40,
565 					y:sNode.point.right.y -10
566 				}
567 				allPoint.push(tmpPoint);
568 				
569 				tmpPoint = {
570 					x:sNode.point.right.x + 40,
571 					y:sNode.point.right.y + 10
572 				}
573 				allPoint.push(tmpPoint);
574 				
575 				tmpPoint = {
576 					x:sNode.point.right.x ,
577 					y:sNode.point.right.y + 10
578 				}
579 				allPoint.push(tmpPoint);
580 			}else{
581 				//非自返情况
582 				setPoint(sNode,dx,dy);
583 				//中间的所有点
584 				for(var i=0;lineArr && i<lineArr.length;i++){
585 					allPoint.push(lineArr[i]);
586 				}
587 				//取最后一个点和终点进行比较
588 				var lastPoint = allPoint[allPoint.length -1];
589 				var dx = lastPoint.x - eNode.point.center.x;
590 				var dy = lastPoint.y - eNode.point.center.y;
591 				setPoint(eNode,dx,dy);
592 			}
593 			
594 			
595 			
596 			
597 			//点准备好了,下面依据所有的点,开始绘制折线
598 			ctx.beginPath();
599 			allGraphInfo.push("ctx.beginPath()");
600 			ctx.moveTo(allPoint[0].x,allPoint[0].y);
601 			allGraphInfo.push("ctx.moveTo("+allPoint[0].x+","+allPoint[0].y+")");
602 			for (var i=1;i<allPoint.length;i++){

723 					//线选的扩散范围值，比如是3，那么就是说，这个线向外扩张3个像素的巨型都是这个线的范围
724 					var ddpx=6;

739 						ctx.lineTo(e.x,e.y-ddpx);

741 					}
